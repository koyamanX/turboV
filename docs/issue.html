ALU Instruction

Instructions which uses ALU are issued to ALU pipes. (e.g. Opcodes variants of OP, OP_IMM, AUIPC, LUI, JAL, JALR)

 1if((!ROB.full) && (!RS.full)) {
 2        if(opcode == LUI) {
 3                Vj = 0;
 4                Qj = 0;
 5        } else if(opcode == JAL || AUIPC) {
 6                Vj = PC;
 7                Qj = 0;
 8        } else {
 9                if(rs1 == 0) {
10                        Vj = 0;
11                        Qj = 0;
12                } else if(register_status_table[rs1] == BUSY) {
13                        if(ROB.read_operandA(register_map_table[rs1]) == BUSY) {
14                                Vj = 0;
15                                Qj = register_map_table[rs1];
16                        } else {
17                                Vj = ROB.read_oprandA_value;
18                                Qj = 0;
19                        }
20                } else {
21                        Vj = GPR.readA(rs1);
22                        Qj = 0;
23                }
24        }
25
26        if(opcode == OP) {
27                if(rs2 == 0) {
28                        Vk = 0;
29                        Qk = 0;
30                } else if(register_status_table[rs2] == BUSY) {
31                        if(ROB.read_operandB(register_map_table[rs2] == BUSY) {
32                                Vk = 0;
33                                Qk = register_map_table[rs2];
34                        } else {
35                                Vk = ROB.read_operandB_value;
36                                Qk = 0;
37                        }
38                } else {
39                        Vk = GPR.readB(rs2);
40                        Qk = 0;
41                }
42        } else {
43                Vk = imm;
44                Qk = 0;
45        }
46
47        if(rd != 0) {
48                register_map_table[rd] := ROB.tag;
49                register_status_table[rd] = BUSY;
50        }
51
52        RS.Busy = 1;
53        RS.Op = Inst.Opcode;
54        RS.Vj = Vj;
55        RS.Vk = Vk;
56        RS.Qj = Qj;
57        RS.Qk = Qk;
58        RS.Dest = ROB.tag
59        RS.A = 0;
60        ROB.Valid = 1;
61        ROB.Busy = 1;
62        ROB.PC = PC:
63        ROB.Inst = Inst;
64        ROB.Dest = rd;
65        // ROB.Value
66        if(opcode == JAL || opcode == JALR) {
67                ROB.Jump = 1;
68        } else {
69                ROB.Jump = 0;
70        }
71        ROB.Target = PC + imm;
72}

Instructions with opcode == BRANCH are issued to Branch pipes.

 2        if(rs1 == 0) {
 3                Vj = 0;

16        }
17

19                Vk = 0;
20                Qk = 0;
21        } else if(register_status_table[rs2] == BUSY) {
22                if(ROB.read_operandB(register_map_table[rs2] == BUSY) {
23                        Vk = 0;
24                        Qk = register_map_table[rs2];
25                } else {
26                        Vk = ROB.read_operandB_value;

33

37        }
38
39        RS.Busy = 1;
40        RS.Op = Inst.Opcode;
41        RS.Vj = Vj;
42        RS.Vk = Vk;

44        RS.Qk = Qk;
45        RS.Dest = ROB.tag
46        RS.A = 0;
47        ROB.Valid = 1;
48        ROB.Busy = 1;
49        ROB.PC = PC:
50        ROB.Inst = Inst;
51        ROB.Dest = rd;
52        // ROB.Value
53        ROB.Jump = 0
54        ROB.Target = PC + imm;
55}
